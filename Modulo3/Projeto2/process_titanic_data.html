def acerta_indices(label):
    """
    def acerta_indices(label: string):

    Funo para realizar o acerto dos ndices de uma determinada coluna de um dataset
    Argumentos:
        label: nome da coluna que ter o ndice ajustado
    Retorna:
        No tem retorno
    """
    if (df[label].dtype == "object"):
        df[label].fillna('', inplace=True)
    else:
        mean = df[label].mean()
        df[label].fillna(mean, inplace=True)
